def decorator(*args, **kwargs):
    def wrapper(fun):
        def inner(*inner_args, **inner_kwargs):
            return fun(*inner_args, **inner_kwargs)
        return inner
    if len(args) == 1 and len(kwargs) == 0 and callable(p := args[0]):
        return wrapper(p)
    return wrapper
